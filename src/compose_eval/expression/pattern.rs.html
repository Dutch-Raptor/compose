29impl Eval for ast::IsExpression<'_> {
30    fn eval(self, vm: &mut Machine) -> SourceResult<Evaluated> {
31        trace_fn!("eval_is_expression");
32        let vm = &mut vm.in_flow_scope_guard();
33        let expr = self.expr();
34        let value = expr.eval(vm)?.value;
35
36        let pat = self.pattern();
37
38        let matched = match destructure_into_flow(vm, value, pat)? {
39            PatternMatchResult::NotMatched(_) => false,
40            PatternMatchResult::Matched => true,
41        };
42
43        Ok(Evaluated::immutable(matched.into_value()))
44    }
45}

47pub fn destructure_into_flow(
48    vm: &mut Machine,
49    value: Value,
50    pat: Pattern,
51) -> SourceResult<PatternMatchResult> {
52    destructure_impl(
53        vm,
54        pat,
55        value,
56        PatternContext::FlowBinding,
57        MatchPath::new(),
58        &mut |vm, expr, value| {
59            match expr {
60                Expr::Ident(ident) => {
61                    let name = ident.get();
62                    let spanned = value
63                        .named(Spanned::new(name.clone(), ident.span()))
64                        // Now that the names have been added, make sure any deferred errors are resolved
65                        .resolved()?;
66
67                    vm.scopes_mut()
68                        .top_flow_mut()
69                        .expect("running in a flow")
70                        .try_bind(name.clone(), Binding::new(spanned, expr.span()))
71                        .at(expr.span())?;
72
73                    Ok(())
74                }
75                _ => bail!(expr.span(), "cannot destructure pattern",),
76            }
77        },
78    )
79}

143pub fn destructure_pattern(
144    vm: &mut Machine,
145    pattern: Pattern,
146    value: Value,
147    ctx: PatternContext,
148    binding_kind: BindingKind,
149    visibility: Visibility,
150) -> SourceResult<PatternMatchResult> {
151    trace_fn!("eval_destructure_pattern");
152    destructure_impl(
153        vm,
154        pattern,
155        value,
156        ctx,
157        MatchPath::new(),
158        &mut |vm, expr, value| match expr {
159            Expr::Ident(ident) => {
160                let name = ident.get().clone();
161                let spanned = value
162                    .named(Spanned::new(name, ident.span()))
163                    // Now that the names have been added, make sure any deferred errors are resolved
164                    .resolved()?;
165
166                vm.define(ident, spanned, binding_kind, visibility)?;
167
168                Ok(())
169            }
170            _ => bail!(expr.span(), "cannot destructure pattern",),
171        },
172    )
173}

175pub fn destructure_impl(
176    vm: &mut Machine,
177    pattern: Pattern,
178    value: Value,
179    ctx: PatternContext,
180    match_path: MatchPath,
181    bind: &mut impl Fn(&mut Machine, Expr, Value) -> SourceResult<()>,
182) -> SourceResult<PatternMatchResult> {
183    trace_fn!("eval_destructure_impl");
184    match pattern {
185        Pattern::Single(expr) => {
186            if let Some(ident) = expr.cast::<Ident>()
187                && let Some(expected_ty) = get_type(vm, ident)
188            {
189                if &value.ty() != expected_ty {
190                    return Ok(PatternMatchResult::NotMatched(
191                        error!(ident.span(), "Type pattern does not match";
192                    note: "expected `{}`, found `{}`", expected_ty, value.ty()),
193                    ));
194                }
195            }
196
197            bind(vm, expr, value)?;
198            Ok(PatternMatchResult::Matched)
199        }
200        Pattern::TypedPattern(typed_pattern) if ctx == PatternContext::LetBinding => {
201            // Types in Compose are resolved at runtime, not during parsing.
202            // This means `let x 1` is initially parsed as a typed pattern (`x 1`).
203            // We resolve the ambiguity here:
204            //   - if `x` resolves to a type → disallowed typed pattern
205            //   - otherwise → assume a missing `=` and emit E0007
206            let ty = typed_pattern.ty();
207            let is_type = get_type(vm, ty).is_some();
208            if is_type {
209                let pat_text = typed_pattern.pattern().to_text();
210                let ty_text = ty.get();
211                bail!(
212                    typed_pattern.span(),
213                    "type patterns are not allowed in `let` bindings";
214                    note: "`let {ty_text} {pat_text} = ...` is not a type annotation in Compose";
215                    hint: "use a `match` or `is` expression to check the type of a value";
216                    code: &E0304_TYPE_PATTERNS_NOT_ALLOWED_IN_LET_BINDINGS;
217                );
218            } else {
219                let binding_text = ty.get();
220                bail!(ty.span().after(), "expected `=` after binding name";
221                    label_message: "expected `=` here";
222                    hint: "if you meant to initialise the binding, add `=`: `let {binding_text} = ...;`";
223                    hint: "or, if you meant to leave it uninitialised, add a semicolon `let {binding_text};`";
224                    code: &E0007_MISSING_EQUALS_AFTER_LET_BINDING)
225            }
226        }
227        Pattern::TypedPattern(typed_pattern) => {
228            let ty_ident = typed_pattern.ty();
229
230            let expected_ty = get_type(vm, ty_ident);
231            let Some(expected_ty) = expected_ty else {
232                bail!(typed_pattern.ty().span(), "Type in typed pattern does not exist";)
233            };
234
235            if &value.ty() != expected_ty {
236                return Ok(PatternMatchResult::NotMatched(
237                    error!(ty_ident.span(), "Type pattern does not match";
238                    note: "expected `{}`, found `{}`", expected_ty, value.ty()),
239                ));
240            }
241
242            destructure_impl(vm, typed_pattern.pattern(), value, ctx, match_path, bind)
243        }
244        Pattern::PlaceHolder(_) => Ok(PatternMatchResult::Matched), // A placeholder means we discard the value, no need to bind
245        Pattern::Destructuring(destruct) => match value {
246            Value::Array(value) => destructure_array(vm, destruct, value, ctx, match_path, bind),
247            Value::Map(value) => destructure_map(vm, destruct, value, ctx, match_path, bind),
248            _ => bail!(pattern.span(), "cannot destructure {}", value.ty()),
249        },
250        Pattern::LiteralPattern(lit) if ctx == PatternContext::LetBinding => bail!(
251            lit.span(),
252            "literal patterns are not allowed in `let` bindings";
253            note: "a `let` binding must introduce at least one variable"
254        ),
255        Pattern::LiteralPattern(lit) if ctx == PatternContext::ForLoopBinding => bail!(
256            lit.span(),
257            "literal patterns are not allowed in `let` bindings";
258            note: "a `for` llop binding must introduce at least one variable"
259        ),
260        Pattern::LiteralPattern(lit) => {
261            let literal_value = lit.eval(vm)?.value;
262
263            if literal_value.not_equals(&value, &vm.heap).at(lit.span())? {
264                return Ok(PatternMatchResult::NotMatched(error!(
265                    lit.span(),
266                    "literal pattern did not match";
267                    label_message: "expected `{}`, got `{}`", literal_value.debug_repr(vm), value.debug_repr(vm);
268                    note: "while matching the value at path {}", match_path;
269                )));
270            }
271
272            Ok(PatternMatchResult::Matched)
273        }
274    }
275}

277fn get_type<'a>(vm: &mut Machine<'a>, ty_ident: Ident) -> Option<&'a Type> {
278    match vm
279        .engine
280        .world
281        .library()
282        .global
283        .scope()
284        .get(ty_ident.get())
285        .map(|b| b.read())?
286    {
287        Value::Type(t) => Some(t),
288        _ => None,
289    }
290}

292fn destructure_array(
293    vm: &mut Machine,
294    destruct: ast::Destructuring,
295    value: ArrayValue,
296    ctx: PatternContext,
297    match_path: MatchPath,
298    bind: &mut impl Fn(&mut Machine, Expr, Value) -> SourceResult<()>,
299) -> SourceResult<PatternMatchResult> {
300    let arr = value.heap_ref().get_unwrap(&vm.heap).clone();
301
302    let len = arr.len();
303    let mut index = 0;
304
305    for p in destruct.items() {
306        match p {
307            DestructuringItem::Pattern(pat) => {
308                let Some(v) = arr.get(index) else {
309                    return Ok(PatternMatchResult::NotMatched(wrong_number_of_elements(
310                        destruct,
311                        len,
312                        &match_path,
313                    )));
314                };
315
316                let matched = destructure_impl(
317                    vm,
318                    pat,
319                    v.clone(),
320                    ctx,
321                    match_path
322                        .clone()
323                        .with_segment(MatchPathSegment::ArrayIndex(index)),
324                    bind,
325                )?;
326
327                if let PatternMatchResult::NotMatched(err) = matched {
328                    return Ok(PatternMatchResult::NotMatched(err));
329                }
330                index += 1;
331            }
332            DestructuringItem::Named(named) => {
333                return Ok(PatternMatchResult::NotMatched(error!(
334                    named.span(),
335                    "cannot destructure a named pattern from an array"
336                )));
337            }
338            DestructuringItem::Spread(spread) => {
339                // The number of elements that have not been bound by a destructuring item and will be bound by the spread
340                let sink_size = (1 + len).checked_sub(destruct.items().count());
341
342                // The items that will be bound by the spread
343                let sunk_items = sink_size.and_then(|n| arr.get(index..index + n));
344
345                let (Some(sink_size), Some(sunk_items)) = (sink_size, sunk_items) else {
346                    return Ok(PatternMatchResult::NotMatched(wrong_number_of_elements(
347                        destruct,
348                        len,
349                        &match_path,
350                    )));
351                };
352
353                if let Some(expr) = spread.sink_expr() {
354                    let sunk_arr =
355                        ArrayValue::from(vm.heap_mut(), sunk_items.to_vec()).into_value();
356                    bind(vm, expr, sunk_arr)?;
357                }
358                index += sink_size;
359            }
360        }
361    }
362
363    // require all items to be bound
364    if index != len {
365        return Ok(PatternMatchResult::NotMatched(wrong_number_of_elements(
366            destruct,
367            len,
368            &match_path,
369        )));
370    }
371
372    Ok(PatternMatchResult::Matched)
373}

375#[allow(unused)]
376fn destructure_map(
377    vm: &mut Machine,
378    destruct: ast::Destructuring,
379    value: MapValue,
380    ctx: PatternContext,
381    match_path: MatchPath,
382    bind: &mut impl Fn(&mut Machine, Expr, Value) -> SourceResult<()>,
383) -> SourceResult<PatternMatchResult> {
384    let map = value.heap_ref().get_unwrap(&vm.heap).clone();
385
386    let mut used: HashSet<&str> = HashSet::new();
387    let mut spread = None;
388
389    for p in destruct.items() {
390        match p {
391            DestructuringItem::Named(named) => {
392                let name = named.name().get();
393                used.insert(name);
394
395                let Some(v) = map.get(name) else {
396                    return Ok(PatternMatchResult::NotMatched(missing_key_err(
397                        name,
398                        named.span(),
399                        &match_path,
400                    )));
401                };
402
403                let matched = destructure_impl(
404                    vm,
405                    named.pattern(),
406                    v.clone(),
407                    ctx,
408                    match_path
409                        .clone()
410                        .with_segment(MatchPathSegment::MapKey(name.clone())),
411                    bind,
412                )?;
413                if let PatternMatchResult::NotMatched(err) = matched {
414                    return Ok(PatternMatchResult::NotMatched(err));
415                }
416            }
417            DestructuringItem::Pattern(Pattern::Single(Expr::Ident(ident))) => {
418                used.insert(ident.get());
419                let Some(v) = map.get(ident.get()) else {
420                    return Ok(PatternMatchResult::NotMatched(missing_key_err(
421                        ident.get(),
422                        ident.span(),
423                        &match_path,
424                    )));
425                };
426
427                bind(vm, Expr::Ident(ident.clone()), v.clone())?;
428            }
429            DestructuringItem::Spread(s) => {
430                spread = Some(s);
431            }
432            DestructuringItem::Pattern(_) => {
433                let mut err = error!(
434                    destruct.span(),
435                    "cannot destructure an unnamed pattern from a map"
436                );
437
438                if !match_path.is_empty() {
439                    err.note(eco_format!(
440                        "while matching the pattern at path {}",
441                        match_path
442                    ))
443                }
444
445                return Ok(PatternMatchResult::NotMatched(err));
446            }
447        }
448    }
449
450    if let Some(spread) = spread
451        && spread.sink_expr().is_some()
452    {
453        let map = HashMap::from_iter(
454            map.iter()
455                .filter(|(k, _)| !used.contains(k.as_str()))
456                .map(|(k, v)| (k.clone(), v.clone())),
457        );
458        let value = MapValue::from(vm.heap_mut(), map).into_value();
459        bind(vm, spread.sink_expr().unwrap(), value)?;
460    }
461
462    if spread.is_none() && used.len() != map.len() {
463        let key_hint = {
464            let missing_keys = map
465                .keys()
466                .filter(|k| !used.contains(k.as_str()))
467                .map(|k| k.as_str())
468                .collect::<Vec<_>>()
469                .tap_mut(|v| v.sort_unstable());
470            format_keyset(&missing_keys)
471        };
472        let mut err = error!(
473            destruct.span(),
474            "map destructuring does not cover all keys";
475            label_message: "unmatched keys: {key_hint}";
476            hint: "add `..` to ignore the remaining keys";
477            hint: "or use `..name` to bind them into a map";
478            code: &E0302_MAP_DESTRUCTURING_UNCOVERED_KEYS,
479        );
480
481        if !match_path.is_empty() {
482            err.note(eco_format!(
483                "while matching the pattern at path {}",
484                match_path
485            ))
486        }
487
488        return Ok(PatternMatchResult::NotMatched(err));
489    }
490
491    Ok(PatternMatchResult::Matched)
492}

494fn missing_key_err(name: &str, span: Span, match_path: &MatchPath) -> SourceDiagnostic {
495    let mut err = error!(
496        span, "missing key in map pattern";
497        label_message: "key `{}` is not present in the map", name;
498        code: &E0303_MAP_DESTRUCTURING_MISSING_KEY_IN_VALUE;
499    );
500
501    if !match_path.is_empty() {
502        err.note(eco_format!(
503            "while matching the pattern at path {}",
504            match_path
505        ))
506    }
507
508    err
509}

511/// Returns a string describing a keyset. lists the first 3 keys, and lists the count of remaining keys.
512fn format_keyset(missing_keys: &[&str]) -> EcoString {
513    const LISTED_KEYS_NUM: usize = 3;
514    let named_keys = missing_keys
515        .iter()
516        .take(LISTED_KEYS_NUM)
517        .copied()
518        .collect::<Vec<_>>()
519        .join(", ");
520    let rest_len = missing_keys.len().saturating_sub(LISTED_KEYS_NUM);
521
522    let more = if rest_len > 0 {
523        &eco_format!(", and {rest_len} more")
524    } else {
525        ""
526    };
527    eco_format!("{}{}", named_keys, more)
528}

530/// Returns a diagnostic indicating that the number of elements in the array does not match the number of elements in the destructuring pattern.
531#[cold]
532fn wrong_number_of_elements(
533    destruct: ast::Destructuring,
534    len: usize,
535    match_path: &MatchPath,
536) -> SourceDiagnostic {
537    let mut count = 0;
538    let mut spread = false;
539
540    for p in destruct.items() {
541        match p {
542            DestructuringItem::Pattern(_) => count += 1,
543            DestructuringItem::Spread(_) => spread = true,
544            DestructuringItem::Named(_) => {}
545        }
546    }
547
548    let quantifier = if len > count {
549        "too many"
550    } else {
551        "not enough"
552    };
553    let expected = match (spread, count) {
554        (true, 1) => "at least 1 element".into(),
555        (true, c) => eco_format!("at least {c} elements"),
556        (false, 0) => "an empty array".into(),
557        (false, 1) => "a single element".into(),
558        (false, c) => eco_format!("{c} elements",),
559    };
560
561    let mut err = error!(
562        destruct.span(), "{quantifier} elements to destructure";
563        hint: "the provided array has a length of {len}, \
564               but the pattern expects {expected}";
565        code: &E0301_ARRAY_DESTRUCTURING_WRONG_NUMBER_OF_ELEMENTS;
566    );
567
568    if len > count {
569        err.hint("use `..` to ignore the remaining elements, or `..name` to bind them");
570    }
571
572    if !match_path.is_empty() {
573        err.note(eco_format!(
574            "while matching the pattern at path {}",
575            match_path
576        ))
577    }
578
579    err
580}
