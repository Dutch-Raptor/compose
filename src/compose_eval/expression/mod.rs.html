29impl Eval for Expr<'_> {
30    fn eval(self, vm: &mut Machine) -> SourceResult<Evaluated> {
31        let span = self.span();
32        let v = match self {
33            Expr::Int(i) => i.eval(vm),
34            Expr::Binary(b) => b.eval(vm),
35            Expr::Ident(i) => i.eval(vm),
36            Expr::CodeBlock(c) => c.eval(vm),
37            Expr::Unit(_) => Ok(Evaluated::unit()),
38            Expr::Str(s) => s.eval(vm),
39            Expr::Unary(u) => u.eval(vm),
40            Expr::Bool(b) => b.eval(vm),
41            Expr::FuncCall(f) => f.eval(vm),
42            Expr::FieldAccess(f) => f.eval(vm),
43            Expr::PathAccess(p) => p.eval(vm),
44            Expr::Parenthesized(p) => p.eval(vm),
45            Expr::Conditional(c) => c.eval(vm),
46            Expr::WhileLoop(w) => w.eval(vm),
47            Expr::ForLoop(f) => f.eval(vm),
48            Expr::Array(a) => a.eval(vm),
49            Expr::Range(r) => r.eval(vm),
50            Expr::Map(m) => m.eval(vm),
51            Expr::Lambda(l) => l.eval(vm),
52            Expr::IndexAccess(i) => i.eval(vm),
53            Expr::MatchExpression(m) => m.eval(vm),
54            Expr::IsExpression(i) => i.eval(vm),
55        }?
56        .spanned(span);
57
58        Ok(v)
59    }
60}
