9/// Visits a closure and determines which variables are captured implicitly.
10#[derive(Debug)]
11pub struct CapturesVisitor<'a> {
12    /// The external scope that variables might be captured from.
13    external: &'a Scopes<'a>,
14    /// The internal scope of variables defined within the closure.
15    internal: Scopes<'a>,
16    /// The variables that are captured.
17    captures: HashMap<EcoString, Span>,
18}
19
20impl<'a> CapturesVisitor<'a> {
21    pub fn new(external: &'a Scopes<'a>, library: Option<&'a Library>, existing: &Scope) -> Self {
22        let mut visitor = Self {
23            external,
24            internal: Scopes::new(library),
25            captures: HashMap::new(),
26        };
27
28        for (k, v) in existing.bindings() {
29            visitor
30                .internal
31                .top_lexical_mut()
32                .bind(k.clone(), v.clone());
33        }
34
35        visitor
36    }
37    pub(crate) fn visit_lambda(&mut self, closure: ast::Lambda<'a>) {
38        for param in closure.params().children() {
39            match param.kind() {
40                ParamKind::Pos(pat) => {
41                    for ident in pat.bindings() {
42                        self.bind(ident);
43                    }
44                }
45                ParamKind::Named(named) => {
46                    self.bind(named.name());
47                }
48            }
49        }
50
51        for capture in closure.captures().children() {
52            self.bind(capture.binding());
53        }
54
55        for statement in closure.statements() {
56            self.visit_statement(statement);
57        }
58    }
59
60    fn visit_statement(&mut self, statement: Statement<'a>) {
61        self.internal.enter_flow();
62        match statement {
63            Statement::Let(let_binding) => {
64                if let Some(init) = let_binding.initial_value() {
65                    self.visit_expr(init)
66                }
67                for binding in let_binding.pattern().bindings() {
68                    self.bind(binding);
69                }
70            }
71            Statement::Expr(expr) => self.visit_expr(expr),
72            Statement::Assign(assign) => {
73                self.visit_expr(assign.lhs());
74                self.visit_expr(assign.rhs());
75            }
76            Statement::Break(brk) => {
77                if let Some(expr) = brk.value() {
78                    self.visit_expr(expr);
79                }
80            }
81            Statement::Return(ret) => {
82                if let Some(expr) = ret.value() {
83                    self.visit_expr(expr);
84                }
85            }
86            Statement::Continue(_) => {}
87            Statement::ModuleImport(import) => {
88                if let Some(alias) = import.alias() {
89                    self.bind(alias);
90                } else {
91                    let path = PathBuf::from(import.source().as_str());
92                    if let Some(stem) = path.file_stem() {
93                        self.bind_eco_str(stem.to_string_lossy().into(), import.source_span());
94                    }
95                }
96            }
97        }
98        self.internal.exit_flow();
99    }
100
101    fn visit_expr(&mut self, expr: Expr<'a>) {
102        match expr {
103            Expr::Ident(ident) => self.capture(ident),
104            Expr::CodeBlock(block) => self.visit_code_block(block),
105            Expr::FieldAccess(access) => {
106                self.visit_expr(access.target());
107            }
108            Expr::Lambda(closure) => {
109                for param in closure.params().children() {
110                    if let ParamKind::Named(named) = param.kind() {
111                        self.visit_expr(named.expr());
112                    }
113                }
114
115                for capture in closure.captures().children() {
116                    self.capture(capture.binding());
117                }
118
119                // NOTE: For now we do not try to analyse the body of the closure.
120                // This is because the closure might try to recursively call itself
121                // and in simple ast walking, that is really hard to resolve correctly.
122                // Any capture errors in the body will be caught when the outer body is evaluated.
123            }
124
125            Expr::ForLoop(for_loop) => {
126                // Created in outer scope
127                self.internal.enter_flow();
128                self.visit_expr(for_loop.iterable());
129
130                self.internal.enter_lexical();
131                let pattern = for_loop.binding();
132                for ident in pattern.bindings() {
133                    self.bind(ident);
134                }
135
136                for stmt in for_loop.body().statements() {
137                    self.visit_statement(stmt);
138                }
139
140                self.internal.exit_lexical();
141                self.internal.exit_flow();
142            }
143            Expr::IsExpression(is) => {
144                self.visit_expr(is.expr());
145
146                for ident in is.pattern().bindings() {
147                    self.bind_flow(ident);
148                }
149            }
150            Expr::Unary(unary) => self.visit_expr(unary.expr()),
151            Expr::Unit(_) => {}
152            Expr::Binary(binary) => {
153                self.visit_expr(binary.lhs());
154                self.visit_expr(binary.rhs());
155            }
156            Expr::Int(_) => {}
157            Expr::Str(_) => {}
158            Expr::Bool(_) => {}
159            Expr::FuncCall(call) => {
160                self.visit_expr(call.callee());
161                for arg in call.args().items() {
162                    match arg {
163                        Arg::Pos(expr) => self.visit_expr(expr),
164                        Arg::Named(named) => self.visit_expr(named.expr()),
165                    }
166                }
167            }
168            Expr::PathAccess(path) => self.visit_expr(path.target()),
169            Expr::Parenthesized(paren) => self.visit_expr(paren.expr()),
170            Expr::Conditional(cond) => {
171                self.internal.enter_flow();
172                self.visit_expr(cond.condition().expr());
173                self.visit_code_block(cond.consequent());
174                self.internal.exit_flow();
175
176                for alternates in cond.cond_alternates() {
177                    self.internal.enter_flow();
178                    self.visit_expr(alternates.condition().expr());
179                    self.visit_code_block(alternates.consequent());
180                    self.internal.exit_flow();
181                }
182
183                if let Some(else_) = cond.cond_else() {
184                    self.visit_code_block(else_.consequent())
185                }
186            }
187            Expr::WhileLoop(while_) => {
188                self.internal.enter_flow();
189                self.visit_expr(while_.condition().expr());
190                self.visit_code_block(while_.body());
191                self.internal.exit_flow();
192            }
193            Expr::Array(arr) => {
194                for expr in arr.elements() {
195                    self.visit_expr(expr);
196                }
197            }
198            Expr::Range(r) => {
199                if let Some(start) = r.start() {
200                    self.visit_expr(start);
201                }
202                if let Some(end) = r.end() {
203                    self.visit_expr(end);
204                }
205            }
206            Expr::Map(map) => {
207                for entry in map.entries() {
208                    match entry.key() {
209                        Expr::Ident(_) => {}
210                        expr => self.visit_expr(expr),
211                    }
212                    self.visit_expr(entry.value());
213                }
214            }
215            Expr::IndexAccess(index) => {
216                self.visit_expr(index.target());
217                self.visit_expr(index.index());
218            }
219            Expr::MatchExpression(match_) => {
220                self.internal.enter_flow();
221                self.visit_expr(match_.expr());
222
223                for arm in match_.match_arms() {
224                    self.internal.enter_flow();
225                    for binding in arm.patterns().flat_map(|pat| pat.bindings()) {
226                        self.bind(binding);
227                    }
228                    if let Some(match_guard) = arm.guard() {
229                        self.visit_expr(match_guard);
230                    }
231                    self.visit_expr(arm.expr());
232                    self.internal.exit_flow();
233                }
234
235                self.internal.exit_flow();
236            }
237        }
238    }
239
240    fn visit_code_block(&mut self, block: ast::CodeBlock<'a>) {
241        self.internal.enter_lexical();
242        for child in block.statements() {
243            self.visit_statement(child);
244        }
245        self.internal.exit_lexical();
246    }
247
248    fn bind(&mut self, ident: Ident) {
249        self.internal.top_lexical_mut().bind(
250            ident.get().clone(),
251            Binding::new(Value::unit(), ident.span()),
252        );
253    }
254
255    fn bind_eco_str(&mut self, name: EcoString, span: Span) {
256        self.internal
257            .top_lexical_mut()
258            .bind(name, Binding::new(Value::unit(), span));
259    }
260
261    fn bind_flow(&mut self, ident: Ident) {
262        if let Some(flow) = self.internal.top_flow_mut() {
263            flow.bind(
264                ident.get().clone(),
265                Binding::new(Value::unit(), ident.span()),
266            );
267        }
268    }
269
270    fn capture(&mut self, ident: Ident<'a>) {
271        if self.internal.get(&ident).is_ok() {
272            // Was defined internally, no need to capture
273            return;
274        }
275
276        // If the value does not exist in the external scope, it is not captured.
277        if self.external.get(&ident).is_ok() {
278            self.captures
279                .entry(ident.get().clone())
280                .or_insert(ident.span());
281        }
282    }
283
284    pub(crate) fn finish(self) -> HashMap<EcoString, Span> {
285        self.captures
286    }
287}

297    #[track_caller]
298    fn test(text: &str, expected_names: &[&str]) {
299        let mut scopes = Scopes::new(None);
300        scopes.top_lexical_mut().define("f", 0i64);
301        scopes.top_lexical_mut().define("x", 0i64);
302        scopes.top_lexical_mut().define("y", 0i64);
303        scopes.top_lexical_mut().define("z", 0i64);
304
305        let mut existing = Scope::new_lexical();
306        existing.define("a", 0i64);
307        existing.define("b", 0i64);
308        existing.define("c", 0i64);
309        let mut visitor = CapturesVisitor::new(&scopes, None, &existing);
310        let world = TestWorld::from_str(text);
311
312
313        let source = world.entrypoint_src();
314        let mut fail = false;
315        for node in source.nodes() {
316            let errors = node
317                .errors()
318                .into_iter()
319                .map(SourceDiagnostic::from)
320                .collect::<Vec<_>>();
321            if !errors.is_empty() {
322                print_diagnostics(&TestWorld::new(), &errors, &[]);
323                fail = true;
324            }
325            visitor.visit_statement(node.cast::<ast::Statement>().expect("expected a statement"));
326        }
327
328        if fail {
329            panic!("failed to parse");
330        }
331
332        let captures = visitor.finish();
333        let mut names: Vec<_> = captures.iter().map(|(k, ..)| k).collect();
334        names.sort();
335
336        assert_eq!(names, expected_names);
337    }
