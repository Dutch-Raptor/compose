17impl Eval for ast::Lambda<'_> {
18    fn eval(self, vm: &mut Machine) -> SourceResult<Evaluated> {
19        let vm = &mut vm.temp_root_guard();
20
21        let mut defaults = Vec::new();
22        for param in self.params().children() {
23            if let ParamKind::Named(named) = param.kind() {
24                defaults.push(named.expr().eval(vm)?.value);
25            }
26        }
27
28        let captured = {
29            let mut errors = EcoVec::new();
30            let mut scope = Scope::new_lexical();
31            for capture in self.captures().children() {
32                let span = capture.binding().span();
33                let name = capture.binding().get();
34                let binding = match vm.get(&capture.binding()).cloned() {
35                    Ok(v) => v,
36                    Err(e) => {
37                        let VariableAccessError::Unbound(unbound) = e else {
38                            errors.push(e.into_source_diagnostic(span));
39                            continue;
40                        };
41
42                        let mut err = error!(
43                            span, "unknown variable `{name}` in closure capture list";
44                            label_message: "variable `{name}` is not defined in the outer scope and cannot be captured";
45                        );
46                        if let Some(msg) = unbound.misspellings_hint_message() {
47                            err.hint(msg);
48                        }
49                        errors.push(err);
50
51                        continue;
52                    }
53                };
54
55                let value = match validate_capture(capture, &binding, vm) {
56                    Ok(v) => v,
57                    Err(e) => {
58                        errors.extend(e.into_iter());
59                        continue;
60                    }
61                };
62
63                scope.bind(
64                    name.clone(),
65                    Binding::new(value.clone(), span).with_kind(match capture.is_mut() {
66                        true => BindingKind::Mutable,
67                        false => BindingKind::Immutable { first_assign: None },
68                    }),
69                );
70            }
71
72            if !errors.is_empty() {
73                return Err(errors);
74            }
75
76            scope
77        };
78
79        let unresolved_captures = {
80            let mut visitor = CapturesVisitor::new(
81                &vm.frames.top.scopes,
82                Some(vm.engine.world.library()),
83                &captured,
84            );
85            visitor.visit_lambda(self);
86            visitor.finish()
87        };
88
89        let closure = Closure {
90            name: None,
91            node: self.to_untyped().clone(),
92            defaults,
93            num_pos_params: self
94                .params()
95                .children()
96                .filter(|p| matches!(p.kind(), ast::ParamKind::Pos(_)))
97                .count(),
98            captured,
99            unresolved_captures,
100        };
101
102        if !vm.context.closure_capture.should_defer() {
103            closure.resolve_captures()?
104        }
105
106        Ok(Func::from(closure)
107            .into_value()
108            .spanned(self.span())
109            .mutable())
110    }
111}

113fn validate_capture<'a>(
114    capture: ast::Capture,
115    binding: &'a Binding,
116    vm: &mut Machine,
117) -> SourceResult<&'a Value> {
118    let span = capture.binding().span();
119    let name = capture.binding().get();
120    if capture.is_mut() && capture.is_ref() && !binding.kind().is_mut() {
121        bail!(
122            capture.span(), "cannot capture variable `{name}` as `ref mut` because it is not declared as mutable";
123            label_message: "capture is declared as a mutable reference";
124            label: Label::secondary(binding.span(), "was defined as immutable here");
125            note: "captured mutable references must match the mutability of the original declaration";
126            hint: "declare the variable as mutable: `let mut {name} = ...`";
127            hint: "or remove `mut` from the capture: `|ref {name}, ...|";
128        );
129    }
130
131    let value = binding.read_checked(span, vm.sink_mut());
132
133    if capture.is_ref() && !value.is_box() {
134        bail!(
135            span, "cannot capture non reference type by reference";
136            label_message: "this captures by reference";
137            note: "only boxed values can be captured by reference"
138        );
139    }
140
141    Ok(value)
142}

144fn define(
145    vm: &mut Machine,
146    ident: Ident,
147    Spanned { value, span }: Spanned<Value>,
148    param: Param,
149) -> SourceResult<()> {
150    if param.is_ref() && !value.is_box() {
151        bail!(
152            ident.span(),
153            "cannot take a reference to a value type. consider boxing it first with `box::new(value)`"
154        );
155    }
156    if !param.is_ref() && value.is_box() {
157        bail!(span, "Cannot bind a box value to a non-reference parameter";
158            label_message: "this parameter is a boxed value";
159            label: Label::primary(ident.span(), "this parameter is declared as an owned value");
160            note: "passing a reference type (like `box`) into an owned parameter is not allowed";
161            hint: "if the parameter intends to modify the original value, mark the parameter as `ref mut`";
162            hint: "if you want to pass the value by ownership, use `.clone_inner()` to create a new copy";
163        );
164    }
165
166    let kind = if param.is_mut() {
167        BindingKind::ParamMut
168    } else {
169        BindingKind::Param
170    };
171    vm.define(ident, value, kind, Visibility::Private)?;
172
173    Ok(())
174}

177pub fn eval_lambda(closure: &Closure, vm: &mut Machine, args: Args) -> SourceResult<Value> {
178    let vm = &mut vm.temp_root_guard();
179    let ast_closure = closure
180        .node
181        .cast::<ast::Lambda>()
182        .expect("closure is not an ast closure");
183    let params = ast_closure.params();
184    let statements = ast_closure.statements();
185
186    // Make sure a gc round is aware that the args are reachable
187    vm.track_tmp_root(&args);
188
189    let result = vm
190        .with_frame(move |vm| {
191            let mut args = args;
192            if let Some(Spanned { value, span }) = &closure.name {
193                vm.try_bind(
194                    value.clone(),
195                    Binding::new(Func::from(closure.clone()), *span),
196                )?;
197            }
198
199            for (k, v) in closure.captured.bindings() {
200                vm.bind(k.clone(), v.clone());
201            }
202
203            let mut defaults = closure.defaults.iter();
204            for p in params.children() {
205                match p.kind() {
206                    ast::ParamKind::Pos(pattern) => match pattern {
207                        Pattern::Single(Expr::Ident(ident)) => {
208                            define(vm, ident, args.expect(&ident)?, p)?;
209                        }
210                        pattern => {
211                            let Some(v) = args.eat()? else {
212                                bail!(pattern.span(), "missing argument for this parameter");
213                            };
214                            let binding_kind = match p.is_mut() {
215                                false => BindingKind::Param,
216                                true => BindingKind::ParamMut,
217                            };
218
219                            match destructure_pattern(
220                                vm,
221                                pattern,
222                                v,
223                                PatternContext::Parameter,
224                                binding_kind,
225                                Visibility::Private,
226                            )? {
227                                PatternMatchResult::NotMatched(err) => bail!(err),
228                                PatternMatchResult::Matched => {}
229                            }
230                        }
231                    },
232                    ast::ParamKind::Named(named) => {
233                        let name = named.name();
234                        let default = defaults.next().unwrap();
235                        let value = args
236                            .named(&name)?
237                            .unwrap_or_else(|| Spanned::new(default.clone(), named.expr().span()));
238                        define(vm, name, value, p)?;
239                    }
240                }
241            }
242
243            // Ensure all args have been used
244            args.finish()?;
245
246            let mut output = Value::unit();
247            for statement in statements {
248                output = statement.eval(vm)?.value;
249                match &vm.flow {
250                    None => {}
251                    Some(FlowEvent::Return(_, Some(explicit))) => {
252                        let explicit = explicit.clone();
253                        vm.flow = None;
254                        return Ok(explicit);
255                    }
256                    Some(FlowEvent::Return(_, None)) => {
257                        vm.flow = None;
258                        return Ok(Value::unit());
259                    }
260                    Some(other) => bail!(other.forbidden()),
261                }
262            }
263
264            SourceResult::Ok(output)
265        })
266        .track_tmp_root(vm);
267
268    vm.maybe_gc();
269
270    result
271}
