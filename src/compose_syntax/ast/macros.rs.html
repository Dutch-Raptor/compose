1/// Implements [`AstNode`] for a struct whose name matches a [`SyntaxKind`]
2/// variant.
3///
4/// The struct becomes a wrapper around a [`SyntaxNode`] pointer, and the
5/// implementation of [`AstNode::from_untyped`] checks that the pointer's kind
6/// matches when converting, returning `Some` or `None` respectively.
7///
8/// The generated struct is the basis for typed accessor methods for properties
9/// of this AST node.
10macro_rules! node {
11    ($(#[$attr:meta])* struct $name:ident) => {
12        // Create the struct as a wrapper around a `SyntaxNode` reference.
13        #[derive(Debug, Copy, Clone, Eq, PartialEq, Hash)]
14        #[repr(transparent)]
15        $(#[$attr])*
16        pub struct $name<'a>(&'a $crate::SyntaxNode);
17
18        impl<'a> $crate::ast::AstNode<'a> for $name<'a> {
19            #[inline]
20            fn from_untyped(node: &'a $crate::SyntaxNode) -> Option<Self> {
21                if node.kind() == $crate::kind::SyntaxKind::$name {
22                    Some(Self(node))
23                } else {
24                    Option::None
25                }
26            }
27
28            #[inline]
29            fn to_untyped(&self) -> &'a $crate::SyntaxNode {
30                self.0
31            }
32        }
33
34        impl Default for $name<'_> {
35            #[inline]
36            fn default() -> Self {
37                static PLACEHOLDER: $crate::SyntaxNode
38                    = $crate::SyntaxNode::placeholder($crate::kind::SyntaxKind::$name);
39                Self(&PLACEHOLDER)
40            }
41        }
42    };
43}
44
45pub(crate) use node;
