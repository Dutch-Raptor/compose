15pub fn match_expr(p: &mut Parser) {
16    trace_fn!("parse_match");
17    let m = p.marker();
18    p.assert(SyntaxKind::MatchKW);
19
20    let wrapped = p.eat_if(SyntaxKind::LeftParen);
21    if !wrapped {
22        p.insert_error_here("condition expressions require parentheses")
23            .with_label_message("Expected an opening `(` before the condition");
24
25        p.recover_until(syntax_set!(LeftBrace));
26
27        // error is not recoverable within the parens, try to recover until after and abort
28        return;
29    }
30
31    if wrapped {
32        code_expression(p);
33        p.expect_closing_delimiter(m, SyntaxKind::RightParen);
34    }
35
36    let open_marker = p.marker();
37    p.expect(SyntaxKind::LeftBrace);
38
39    while !p.current().is_terminator() {
40        // match arm
41        let arm_m = p.marker();
42
43        // patterns
44        let mut first_bindings = HashSet::new();
45        pattern(p, false, false, &mut first_bindings);
46        while p.eat_if(SyntaxKind::Pipe) {
47            let mut current_bindings = HashSet::new();
48            pattern(p, false, false, &mut current_bindings);
49
50            if current_bindings != first_bindings {
51                insert_pattern_bindings_mismatch_error(
52                    p,
53                    &mut first_bindings,
54                    &mut current_bindings,
55                );
56            }
57        }
58
59        // guard
60        if p.eat_if(SyntaxKind::IfKW) {
61            code_expression(p);
62        }
63
64        if !p.eat_if(SyntaxKind::Arrow) {
65            p.insert_error_here("expected `=>` in match arm")
66                .with_label_message("expected `=>` here");
67
68            if !p.at_set(set::EXPR) {
69                // If the `=>` is not missing, just eat a token to make progress
70                p.eat();
71            }
72            continue;
73        }
74
75        code_expression(p);
76
77        p.wrap(arm_m, SyntaxKind::MatchArm);
78
79        if !p.current().is_terminator() {
80            p.expect_or_recover(SyntaxKind::Comma, syntax_set!(RightBracket));
81        }
82    }
83
84    p.expect_closing_delimiter(open_marker, SyntaxKind::RightBrace);
85
86    p.wrap(m, SyntaxKind::MatchExpression)
87}

89fn insert_pattern_bindings_mismatch_error(
90    p: &mut Parser,
91    first_bindings: &mut HashSet<&str>,
92    current_bindings: &mut HashSet<&str>,
93) {
94    let span = p.last_node().unwrap().span();
95
96    let mut introduced = current_bindings.difference(&first_bindings).peekable();
97    let mut missing = first_bindings.difference(&current_bindings).peekable();
98
99    let mut diff_note = eco_format!("pattern bindings mismatch");
100
101    if missing.peek().is_some() {
102        diff_note.push_str(&eco_format!(
103            "\n  - not bound here: `{}`",
104            missing.join("`, `")
105        ))
106    }
107    if introduced.peek().is_some() {
108        diff_note.push_str(&eco_format!(
109            "\n  - bound only here: `{}`",
110            introduced.join("`, `")
111        ));
112    };
113
114    p.insert_error_at(span, "patterns within a match arm have differing bindings")
115        .with_code(&E0311_MATCH_ARM_PATTERNS_BIND_DIFFERENT_VARIABLES)
116        .with_hint("all patterns joined with `|` must bind the same variables, because the arm body must work for every pattern")
117        .with_hint(diff_note);
118}

120/// Parses a binding or reassignment pattern.
121pub fn pattern<'s>(
122    p: &mut Parser<'s>,
123    reassignment: bool,
124    in_typed_pattern: bool,
125    seen: &mut HashSet<&'s str>,
126) {
127    trace_fn!("parse_pattern");
128    match p.current() {
129        // Literals
130        SyntaxKind::Int | SyntaxKind::Float | SyntaxKind::Str | SyntaxKind::Bool => p.eat(),
131        SyntaxKind::LeftParen if p.peek_at(SyntaxKind::RightParen) => {
132            let m = p.marker();
133            p.assert(SyntaxKind::LeftParen);
134            p.assert(SyntaxKind::RightParen);
135            p.wrap(m, SyntaxKind::Unit);
136        }
137
138        SyntaxKind::Underscore => p.eat(),
139        SyntaxKind::LeftBracket => destructure_array(p, seen),
140        SyntaxKind::LeftBrace => destructure_map(p, seen),
141        _ => pattern_leaf(p, seen, reassignment, in_typed_pattern),
142    }
143}

145fn destructure_map<'s>(p: &mut Parser<'s>, seen: &mut HashSet<&'s str>) {
146    trace_fn!("parse_destructure_map");
147    let m = p.marker();
148    p.assert(SyntaxKind::LeftBrace);
149    
150    let mut sink = false;
151    while !p.current().is_terminator() {
152        if !p.at_set(set::DESTRUCTURING_ITEM) {
153            p.unexpected("expected a destructuring item", None);
154            continue;
155        }
156        
157        destructuring_item(p, seen, &mut sink);
158
159        if !p.current().is_terminator() {
160            p.expect_or_recover(SyntaxKind::Comma, syntax_set!(RightBrace));
161        }
162    }
163
164    p.expect_closing_delimiter(m, SyntaxKind::RightBrace);
165
166    p.wrap(m, SyntaxKind::Destructuring);
167}

196fn destructuring_item<'s>(p: &mut Parser<'s>, seen: &mut HashSet<&'s str>, sink: &mut bool) {
197    let m = p.marker();
198
199    if p.eat_if(SyntaxKind::Dots) {
200        // sink
201        if p.at_set(set::PATTERN_LEAF) {
202            pattern_leaf(p, seen, false, false);
203        }
204        p.wrap(m, SyntaxKind::Spread);
205        if mem::replace(sink, true) {
206            p[m].convert_to_error("duplicate spread operator (`..`) in destructuring pattern");
207        }
208        return;
209    }
210
211    // parse normal array element or map key
212    pattern(p, false, false, seen);
213
214    if p.eat_if(SyntaxKind::Colon) {
215        pattern(p, true, false, seen);
216
217        if p[m].kind() != SyntaxKind::Ident {
218            p[m].expected("identifier after `:` in destructuring pattern")
219        }
220
221        p.wrap(m, SyntaxKind::Named)
222    }
223}

225fn pattern_leaf<'s>(p: &mut Parser<'s>, seen: &mut HashSet<&'s str>, reassignment: bool, in_typed_pattern: bool) {
226    trace_fn!("parse_pattern_leaf");
227    if p.current().is_keyword() {
228        p.token.node.expected("pattern");
229        p.eat();
230        return;
231    } else if !p.at_set(set::PATTERN_LEAF) {
232        p.insert_error_here("expected a pattern");
233        return;
234    }
235
236    if p.eat_if(SyntaxKind::Underscore) {
237        return;
238    }
239
240    let m = p.marker();
241    let mut binding_text = p.current_text();
242
243    // Parse a full atomic expression, even though we only care about an identifier.
244    // This way the entire expression can be marked as an error if it is not.
245    expressions::code_expr_prec(p, ExprContext::AtomicExpr, Precedence::Lowest);
246
247    let last_node = p.last_node().expect("was just parsed");
248    let mut span = last_node.span();
249    let last_kind = last_node.kind();
250
251    if !reassignment && last_node.kind() != SyntaxKind::Ident {
252        p.insert_error_at(span, "pattern leaf must be an identifier or `_`")
253            .with_label_message("expected this to be an identifier");
254        return;
255    }
256
257    // Do not allow nested typed bindings
258    let at_typed_binding = last_kind == SyntaxKind::Ident && p.at_set(set::PATTERN) && !in_typed_pattern;
259
260    if at_typed_binding {
261        binding_text = p.current_text();
262        span = p.current_span();
263
264        pattern(p, reassignment, true, seen);
265        p.wrap(m, SyntaxKind::TypedPattern);
266    }
267
268    if !seen.insert(binding_text) && reassignment {
269        p.insert_error_at(span, "duplicate binding")
270            .with_label_message("this binding already appears in this pattern");
271    }
272}
