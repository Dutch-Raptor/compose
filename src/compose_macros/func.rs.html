11pub fn func(stream: TokenStream, item: &ItemFn) -> Result<TokenStream> {
12    let func = parse(stream, item)?;
13    Ok(create(&func, item))
14}

16fn create(func: &Func, item: &ItemFn) -> TokenStream {
17    let Func { rust_name, vis, .. } = func;
18    let definition = rewrite_fn_item(item);
19
20    // Create a type to which the func data will be attached
21    let function_type = create_func_ty(func);
22    let data = create_func_data(func);
23
24    let data_impl = if function_type.is_some() {
25        quote! {
26            impl ::compose_library::foundations::NativeFunc for #rust_name {
27                fn data() -> &'static ::compose_library::foundations::NativeFuncData {
28                    static DATA: #foundations::NativeFuncData = #data;
29                    &DATA
30                }
31            }
32        }
33    } else {
34        let ident_data = quote::format_ident!("{rust_name}_data");
35        quote! {
36            #vis fn #ident_data() -> &'static #foundations::NativeFuncData {
37                static DATA: #foundations::NativeFuncData = #data;
38                &DATA
39            }
40        }
41    };
42
43    quote! {
44        #definition
45        #function_type
46        #data_impl
47    }
48}

50fn rewrite_fn_item(item: &ItemFn) -> ItemFn {
51    let inputs = item.sig.inputs.iter().cloned().filter_map(|mut input| {
52        if let FnArg::Typed(typed) = &mut input {
53            if typed
54                .attrs
55                .iter()
56                .any(|attr| attr.path().is_ident("external"))
57            {
58                return None;
59            }
60            typed.attrs.clear();
61        }
62        Some(input)
63    });
64    let mut item = item.clone();
65    item.attrs.clear();
66    item.sig.inputs = parse_quote! { #(#inputs),* };
67    item
68}

70fn create_func_data(func: &Func) -> TokenStream {
71    let Func {
72        name,
73        rust_name,
74        params: _params,
75        return_type: _return_type,
76        scope,
77        parent: _parent,
78        vis: _vis,
79        special,
80    } = func;
81
82    let scope = if *scope {
83        quote! { <#rust_name as #foundations::NativeScope>::scope() }
84    } else {
85        quote! { &#foundations::EMPTY_SCOPE }
86    };
87
88    let closure = create_wrapper_closure(func);
89
90    let fn_type = match special.self_ {
91        Some(Param { binding: Binding::RefMut, ..}) => quote! { #foundations::FuncType::MethodMut },
92        Some(_) => quote! { #foundations::FuncType::Method },
93        None => quote! { #foundations::FuncType::Associated },   
94    };
95
96    let name = quote! { #name };
97
98    quote! {
99        ::compose_library::foundations::NativeFuncData {
100            name: #name,
101            closure: #closure,
102            scope: ::std::sync::LazyLock::new(|| #scope),
103            fn_type: #fn_type
104        }
105    }
106}

108fn create_wrapper_closure(func: &Func) -> TokenStream {
109    let arg_handlers = {
110        let args = func.params.iter().map(create_param_parser);
111
112        let self_handler = func.special.self_.as_ref().map(create_param_parser);
113
114        quote! {
115            #self_handler
116            #(#args)*
117        }
118    };
119
120    let finish = Some(quote! {args.take().finish()?;  });
121
122    let call = {
123        let self_ = func
124            .special
125            .self_
126            .as_ref()
127            .map(bind)
128            .map(|tokens| quote! { #tokens, });
129        let vm = func.special.vm.then(|| quote! { vm, });
130        let forwarded = func.params.iter().map(bind);
131
132        quote! {
133            __func(#self_ #vm #(#forwarded,)*)
134        }
135    };
136
137    let parent = func.parent.as_ref().map(|ty| quote! { #ty:: });
138    let ident = &func.rust_name;
139    quote! {
140        |vm, args| {
141            let __func = #parent #ident;
142            #arg_handlers
143            #finish
144            let ret = #call;
145            ::compose_library::foundations::IntoResult::into_result(ret, args.span)
146        }
147    }
148}

150fn bind(param: &Param) -> TokenStream {
151    let Param { ident, binding, .. } = param;
152    match binding {
153        Binding::Owned => quote! { #ident },
154        Binding::Ref => quote! { &#ident },
155        Binding::RefMut => quote! { &mut #ident },
156    }
157}

159fn create_param_parser(param: &Param) -> TokenStream {
160    let Param {
161        ident, ty, name, ..
162    } = param;
163
164    let value = if param.variadic {
165        quote! {
166            args.all()?
167        }
168    } else if param.named {
169        quote! {
170            args.named(#name)?
171        }
172    } else {
173        quote! {
174            args.expect(#name)?
175        }
176    };
177
178    quote! {
179        let mut #ident: #ty = #value;
180    }
181}

183fn create_func_ty(func: &Func) -> Option<TokenStream> {
184    if func.parent.is_some() {
185        return None;
186    }
187    let ident = &func.rust_name;
188    Some(quote! {
189        #[doc(hidden)]
190        #[allow(non_camel_case_types)]
191        pub enum #ident {}
192    })
193}

225fn parse(stream: TokenStream, item: &ItemFn) -> Result<Func> {
226    let meta: Meta = syn::parse2(stream)?;
227    let name = determine_name(&item.sig.ident, meta.name);
228
229    let mut params = Vec::new();
230    let mut special = SpecialParams::default();
231    for input_param in &item.sig.inputs {
232        parse_param(&mut special, &mut params, meta.parent.as_ref(), input_param)?;
233    }
234
235    if meta.parent.is_some() && meta.scope {
236        bail!(item, "A function in a scope cannot have a scope")
237    }
238
239    let return_type = match &item.sig.output {
240        ReturnType::Default => parse_quote! { () },
241        ReturnType::Type(_, ty) => *ty.clone(),
242    };
243
244    let rust_name = item.sig.ident.clone();
245
246    Ok(Func {
247        name,
248        rust_name,
249        params,
250        return_type,
251        special,
252        scope: meta.scope,
253        parent: meta.parent,
254        vis: item.vis.clone(),
255    })
256}

258/// The `..` in `#[func(..)]`.
259pub struct Meta {
260    /// Whether this function has an associated scope defined by the `#[scope]` macro.
261    pub scope: bool,
262    /// The function's name as exposed to Compose.
263    pub name: Option<String>,
264    /// The parent type of this function.
265    ///
266    /// Used for functions in a scope.
267    pub parent: Option<Type>,
268}

270impl Parse for Meta {
271    fn parse(input: ParseStream<'_>) -> Result<Self> {
272        Ok(Self {
273            scope: parse_flag::<kw::scope>(input)?,
274            name: parse_string::<kw::name>(input)?,
275            parent: parse_key_value::<kw::parent, _>(input)?,
276        })
277    }

286fn parse_param(
287    special_params: &mut SpecialParams,
288    params: &mut Vec<Param>,
289    parent: Option<&Type>,
290    input: &FnArg,
291) -> Result<()> {
292    let typed = match input {
293        FnArg::Receiver(recv) => {
294            let binding = match (&recv.reference, &recv.mutability) {
295                (None, None) => Binding::Owned,
296                (Some(_), None) => Binding::Ref,
297                (_, Some(_)) => Binding::RefMut,
298            };
299
300            special_params.self_ = Some(Param {
301                binding,
302                ident: Ident::new("self_", recv.self_token.span()),
303                ty: match parent {
304                    Some(ty) => ty.clone(),
305                    None => bail!(
306                        recv,
307                        "explicit parent type is required for functions with a self parameter"
308                    ),
309                },
310                docs: documentation(&recv.attrs),
311                name: "self".to_string(),
312                variadic: false,
313                named: false,
314            });
315            return Ok(());
316        }
317        FnArg::Typed(typed) => typed,
318    };
319
320    let ident = match typed.pat.as_ref() {
321        syn::Pat::Ident(syn::PatIdent { ident, .. }) => ident,
322        _ => bail!(
323            typed.pat,
324            "expected identifier. Destructuring is not supported"
325        ),
326    };
327
328    match ident.to_string().as_str() {
329        "vm" => special_params.vm = true,
330        _ => {
331            let mut attrs = typed.attrs.clone();
332
333            params.push(Param {
334                binding: Binding::Owned,
335                ident: ident.clone(),
336                ty: *typed.ty.clone(),
337                name: ident.to_string(),
338                docs: documentation(&attrs),
339                variadic: has_attr(&mut attrs, "variadic"),
340                named: has_attr(&mut attrs, "named"),
341            });
342        }
343    }
344
345    Ok(())
346}
