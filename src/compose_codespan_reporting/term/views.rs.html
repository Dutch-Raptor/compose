10/// Calculate the number of decimal digits in `n`.
11// TODO: simplify after https://github.com/rust-lang/rust/issues/70887 resolves
12fn count_digits(n: usize) -> usize {
13    // Use a saturating_add because in that edge case the number of digits
14    // will not be changed.
15    (n.saturating_add(1) as f64).log10().ceil() as usize
16}

20struct LabeledFile<'diagnostic, FileId> {
21    file_id: FileId,
22    start: usize,
23    name: String,
24    location: Location,
25    num_multi_labels: usize,
26    lines: BTreeMap<usize, Line<'diagnostic>>,
27    max_label_style: LabelStyle,
28}

30impl<'diagnostic, FileId> LabeledFile<'diagnostic, FileId> {
31    fn get_or_insert_line(
32        &mut self,
33        line_index: usize,
34        line_range: Range<usize>,
35        line_number: usize,
36    ) -> &mut Line<'diagnostic> {
37        self.lines.entry(line_index).or_insert_with(|| Line {
38            range: line_range,
39            number: line_number,
40            single_labels: vec![],
41            multi_labels: vec![],
42            // This has to be false by default so we know if it must be rendered by another condition already.
43            must_render: false,
44        })
45    }
46}

74    fn label_files<'files>(
75        &self,
76        files: &'files impl Files<'files, FileId = FileId>,
77        labels: &'diagnostic [Label<FileId>],
78    ) -> Result<(usize, Vec<LabeledFile<'diagnostic, FileId>>), Error> {
79        // TODO: Make this data structure external, to allow for allocation reuse
80        let mut labeled_files = Vec::<LabeledFile<'_, _>>::new();
81        // Keep track of the outer padding to use when rendering the
82        // snippets of source code.
83        let mut outer_padding = 0;
84
85        // Group labels by file
86        for label in labels {
87            let start_line_index = files.line_index(label.file_id, label.range.start)?;
88            let start_line_number = files.line_number(label.file_id, start_line_index)?;
89            let start_line_range = files.line_range(label.file_id, start_line_index)?;
90            let end_line_index = files.line_index(label.file_id, label.range.end)?;
91            let end_line_number = files.line_number(label.file_id, end_line_index)?;
92            let end_line_range = files.line_range(label.file_id, end_line_index)?;
93
94            outer_padding = std::cmp::max(outer_padding, count_digits(start_line_number));
95            outer_padding = std::cmp::max(outer_padding, count_digits(end_line_number));
96
97            // NOTE: This could be made more efficient by using an associative
98            // data structure like a hashmap or B-tree,  but we use a vector to
99            // preserve the order that unique files appear in the list of labels.
100            let labeled_file = match labeled_files
101                .iter_mut()
102                .find(|labeled_file| label.file_id == labeled_file.file_id)
103            {
104                Some(labeled_file) => {
105                    // another diagnostic also referenced this file
106                    if labeled_file.max_label_style > label.style
107                        || (labeled_file.max_label_style == label.style
108                            && labeled_file.start > label.range.start)
109                    {
110                        // this label has a higher style or has the same style but starts earlier
111                        labeled_file.start = label.range.start;
112                        labeled_file.location = files.location(label.file_id, label.range.start)?;
113                        labeled_file.max_label_style = label.style;
114                    }
115                    labeled_file
116                }
117                None => {
118                    // no other diagnostic referenced this file yet
119                    labeled_files.push(LabeledFile {
120                        file_id: label.file_id,
121                        start: label.range.start,
122                        name: files.name(label.file_id)?.to_string(),
123                        location: files.location(label.file_id, label.range.start)?,
124                        num_multi_labels: 0,
125                        lines: BTreeMap::new(),
126                        max_label_style: label.style,
127                    });
128                    // this unwrap should never fail because we just pushed an element
129                    labeled_files
130                        .last_mut()
131                        .expect("just pushed an element that disappeared")
132                }
133            };
134
135            // insert context lines before label
136            // start from 1 because 0 would be the start of the label itself
137            for offset in 1..self.config.before_label_lines + 1 {
138                let index = if let Some(index) = start_line_index.checked_sub(offset) {
139                    index
140                } else {
141                    // we are going from smallest to largest offset, so if
142                    // the offset can not be subtracted from the start we
143                    // reached the first line
144                    break;
145                };
146
147                if let Ok(range) = files.line_range(label.file_id, index) {
148                    let line =
149                        labeled_file.get_or_insert_line(index, range, start_line_number - offset);
150                    line.must_render = true;
151                } else {
152                    break;
153                }
154            }
155
156            // insert context lines after label
157            // start from 1 because 0 would be the end of the label itself
158            for offset in 1..self.config.after_label_lines + 1 {
159                let index = end_line_index
160                    .checked_add(offset)
161                    .expect("line index too big");
162
163                if let Ok(range) = files.line_range(label.file_id, index) {
164                    let line =
165                        labeled_file.get_or_insert_line(index, range, end_line_number + offset);
166                    line.must_render = true;
167                } else {
168                    break;
169                }
170            }
171
172            if start_line_index == end_line_index {
173                // Single line
174                //
175                // ```text
176                // 2 │ (+ test "")
177                //   │         ^^ expected `Int` but found `String`
178                // ```
179                let label_start = label.range.start - start_line_range.start;
180                // Ensure that we print at least one caret, even when we
181                // have a zero-length source range.
182                let label_end =
183                    usize::max(label.range.end - start_line_range.start, label_start + 1);
184
185                let line = labeled_file.get_or_insert_line(
186                    start_line_index,
187                    start_line_range,
188                    start_line_number,
189                );
190
191                // Ensure that the single line labels are lexicographically
192                // sorted by the range of source code that they cover.
193                let index = match line.single_labels.binary_search_by(|(_, range, _)| {
194                    // `Range<usize>` doesn't implement `Ord`, so convert to `(usize, usize)`
195                    // to piggyback off its lexicographic comparison implementation.
196                    (range.start, range.end).cmp(&(label_start, label_end))
197                }) {
198                    // If the ranges are the same, order the labels in reverse
199                    // to how they were originally specified in the diagnostic.
200                    // This helps with printing in the renderer.
201                    Ok(index) | Err(index) => index,
202                };
203
204                line.single_labels
205                    .insert(index, (label.style, label_start..label_end, &label.message));
206
207                // If this line is not rendered, the SingleLabel is not visible.
208                line.must_render = true;
209            } else {
210                // Multiple lines
211                //
212                // ```text
213                // 4 │   fizz₁ num = case (mod num 5) (mod num 3) of
214                //   │ ╭─────────────^
215                // 5 │ │     0 0 => "FizzBuzz"
216                // 6 │ │     0 _ => "Fizz"
217                // 7 │ │     _ 0 => "Buzz"
218                // 8 │ │     _ _ => num
219                //   │ ╰──────────────^ `case` clauses have incompatible types
220                // ```
221
222                let label_index = labeled_file.num_multi_labels;
223                labeled_file.num_multi_labels += 1;
224
225                // First labeled line
226                let label_start = label.range.start - start_line_range.start;
227
228                let start_line = labeled_file.get_or_insert_line(
229                    start_line_index,
230                    start_line_range.clone(),
231                    start_line_number,
232                );
233
234                start_line.multi_labels.push((
235                    label_index,
236                    label.style,
237                    MultiLabel::Top(label_start),
238                ));
239
240                // The first line has to be rendered so the start of the label is visible.
241                start_line.must_render = true;
242
243                // Marked lines
244                //
245                // ```text
246                // 5 │ │     0 0 => "FizzBuzz"
247                // 6 │ │     0 _ => "Fizz"
248                // 7 │ │     _ 0 => "Buzz"
249                // ```
250                for line_index in (start_line_index + 1)..end_line_index {
251                    let line_range = files.line_range(label.file_id, line_index)?;
252                    let line_number = files.line_number(label.file_id, line_index)?;
253
254                    outer_padding = std::cmp::max(outer_padding, count_digits(line_number));
255
256                    let line = labeled_file.get_or_insert_line(line_index, line_range, line_number);
257
258                    line.multi_labels
259                        .push((label_index, label.style, MultiLabel::Left));
260
261                    // The line should be rendered to match the configuration of how much context to show.
262                    line.must_render |=
263                        // Is this line part of the context after the start of the label?
264                        line_index - start_line_index <= self.config.start_context_lines
265                        ||
266                        // Is this line part of the context before the end of the label?
267                        end_line_index - line_index <= self.config.end_context_lines;
268                }
269
270                // Last labeled line
271                //
272                // ```text
273                // 8 │ │     _ _ => num
274                //   │ ╰──────────────^ `case` clauses have incompatible types
275                // ```
276                let label_end = label.range.end - end_line_range.start;
277
278                let end_line = labeled_file.get_or_insert_line(
279                    end_line_index,
280                    end_line_range,
281                    end_line_number,
282                );
283
284                end_line.multi_labels.push((
285                    label_index,
286                    label.style,
287                    MultiLabel::Bottom(label_end, &label.message),
288                ));
289
290                // The last line has to be rendered so the end of the label is visible.
291                end_line.must_render = true;
292            }
293        }
294        Ok((outer_padding, labeled_files))
295    }

297    /// Source snippets
298    ///
299    /// ```text
300    ///   ┌─ test:2:9
301    ///   │
302    /// 2 │ (+ test "")
303    ///   │         ^^ expected `Int` but found `String`
304    ///   │
305    /// ```
306    fn render_source_snippets<'files>(
307        &self,
308        outer_padding: usize,
309        files: &'files impl Files<'files, FileId = FileId>,
310        labeled_files: Vec<LabeledFile<'diagnostic, FileId>>,
311        renderer: &mut Renderer<'_, '_>,
312    ) -> Result<(), Error> {
313        let mut labeled_files = labeled_files.into_iter().peekable();
314        while let Some(labeled_file) = labeled_files.next() {
315            let source = files.source(labeled_file.file_id)?;
316            let source = source.as_ref();
317
318            // Top left border and locus.
319            //
320            // ```text
321            // ┌─ test:2:9
322            // ```
323            if !labeled_file.lines.is_empty() {
324                renderer.render_snippet_start(
325                    outer_padding,
326                    &Locus {
327                        name: labeled_file.name,
328                        location: labeled_file.location,
329                    },
330                )?;
331                renderer.render_snippet_empty(
332                    outer_padding,
333                    self.diagnostic.severity,
334                    labeled_file.num_multi_labels,
335                    &[],
336                )?;
337            }
338
339            let mut lines = labeled_file
340                .lines
341                .iter()
342                .filter(|(_, line)| line.must_render)
343                .peekable();
344
345            while let Some((line_index, line)) = lines.next() {
346                renderer.render_snippet_source(
347                    outer_padding,
348                    line.number,
349                    &source[line.range.clone()],
350                    self.diagnostic.severity,
351                    &line.single_labels,
352                    labeled_file.num_multi_labels,
353                    &line.multi_labels,
354                )?;
355
356                // Check to see if we need to render any intermediate stuff
357                // before rendering the next line.
358                if let Some((next_line_index, next_line)) = lines.peek() {
359                    match next_line_index.checked_sub(*line_index) {
360                        // Consecutive lines
361                        Some(1) => {}
362                        // One line between the current line and the next line
363                        Some(2) => {
364                            // Write a source line
365                            let file_id = labeled_file.file_id;
366
367                            // This line was not intended to be rendered initially.
368                            // To render the line right, we have to get back the original labels.
369                            let labels = labeled_file
370                                .lines
371                                .get(&(line_index + 1))
372                                .map_or(&[][..], |line| &line.multi_labels[..]);
373
374                            renderer.render_snippet_source(
375                                outer_padding,
376                                files.line_number(file_id, line_index + 1)?,
377                                &source[files.line_range(file_id, line_index + 1)?],
378                                self.diagnostic.severity,
379                                &[],
380                                labeled_file.num_multi_labels,
381                                labels,
382                            )?;
383                        }
384                        // More than one line between the current line and the next line.
385                        Some(_) | None => {
386                            // Source break
387                            //
388                            // ```text
389                            // ·
390                            // ```
391                            renderer.render_snippet_break(
392                                outer_padding,
393                                self.diagnostic.severity,
394                                labeled_file.num_multi_labels,
395                                &next_line.multi_labels,
396                            )?;
397                        }
398                    }
399                }
400            }
401
402            // Check to see if we should render a trailing border after the
403            // final line of the snippet.
404            if labeled_files.peek().is_none()
405                && self.diagnostic.notes.is_empty()
406                && self.diagnostic.subdiagnostics.is_empty()
407            {
408                // We don't render a border if we are at the final newline
409                // without trailing notes, because it would end up looking too
410                // spaced-out in combination with the final new line.
411            } else {
412                // Render the trailing snippet border.
413                renderer.render_snippet_empty(
414                    outer_padding,
415                    self.diagnostic.severity,
416                    labeled_file.num_multi_labels,
417                    &[],
418                )?;
419            }
420        }
421        Ok(())
422    }

424    pub fn render<'files>(
425        &self,
426        files: &'files impl Files<'files, FileId = FileId>,
427        renderer: &mut Renderer<'_, '_>,
428    ) -> Result<(), Error>
429    where
430        FileId: 'files,
431    {
432        let (outer_padding, labeled_files) = self.label_files(files, &self.diagnostic.labels)?;
433        // Header and message
434        //
435        // ```text
436        // error[E0001]: unexpected type in `+` application
437        // ```
438        renderer.render_header(
439            None,
440            self.diagnostic.severity,
441            self.diagnostic.code.as_deref(),
442            self.diagnostic.message.as_str(),
443        )?;
444
445        self.render_source_snippets(outer_padding, files, labeled_files, renderer)?;
446
447        // Additional notes
448        //
449        // ```text
450        // = expected type `Int`
451        //      found type `String`
452        // ```
453        for note in &self.diagnostic.notes {
454            renderer.render_snippet_note(outer_padding, note)?;
455        }
456
457        // Subdiagnostics
458        let mut multiline_suggestions = 0;
459        let mut empty_suggestions = 0;
460        for subdiagnostic in &self.diagnostic.subdiagnostics {
461            match subdiagnostic {
462                // Suggestions
463                //
464                // ```text
465                // = replace size here
466                // | 2       let a: [bool; N] = [true, false];
467                // |                       ~ replace size here
468                // ```
469                // FIXME: We currently skip suggestions that remove or insert multiple lines.
470                Subdiagnostic::Suggestion(Suggestion {
471                    file_id,
472                    message,
473                    parts,
474                }) => {
475                    let file_id = *file_id;
476
477                    if parts.is_empty() {
478                        empty_suggestions += 1;
479                        continue;
480                    }
481
482                    // The start of the first suggestion.
483                    let start_line_offset = parts
484                        .iter()
485                        .map(|part| part.range.start)
486                        .min()
487                        .expect("(internal) should have checked for if no parts exist already");
488                    let start_line_index = files.line_index(file_id, start_line_offset)?;
489                    let start_line_number = files.line_number(file_id, start_line_index)?;
490                    let start_line_range = files.line_range(file_id, start_line_index)?;
491
492                    // The end of the last suggestion.
493                    let end_line_offset = parts
494                        .iter()
495                        .map(|part| part.range.end)
496                        .max()
497                        .expect("(internal) should have checked for if no parts exist already");
498                    let end_line_index = files.line_index(file_id, end_line_offset)?;
499                    let end_line_number = files.line_number(file_id, end_line_index)?;
500                    let end_line_range = files.line_range(file_id, end_line_index)?;
501
502                    let line_number_range = start_line_number..end_line_number;
503                    // Get the lines that include the text to be replaced.
504                    let source = files.source(file_id)?;
505                    let source = &source.as_ref()[start_line_range.start..end_line_range.end];
506
507                    let replacement_has_newline =
508                        parts.iter().any(|part| part.replacement.contains("\n"));
509                    if start_line_number != end_line_number || replacement_has_newline {
510                        multiline_suggestions += 1;
511
512                        // for now, output a normal multi-line label and skip rendering the replacement
513                        let labels =
514                            vec![
515                                Label::secondary(file_id, start_line_offset..end_line_offset)
516                                    .with_message(message),
517                            ];
518                        let (outer_padding, labeled_files) = self.label_files(files, &labels)?;
519                        self.render_source_snippets(outer_padding, files, labeled_files, renderer)?;
520                        continue;
521                    }
522
523                    let parts: Vec<_> = parts
524                        .iter()
525                        .map(|SuggestionPart { range, replacement }| {
526                            // Byte index in `source`.
527                            // NOTE: Not necessarily the correct thinking for multiple lines.
528                            let replacement_start = range.start - start_line_range.start;
529                            let replacement_end = range.end - start_line_range.start;
530                            let replacement_range = replacement_start..replacement_end;
531                            (replacement_range, replacement.as_str())
532                        })
533                        .collect();
534
535                    renderer.render_suggestions(
536                        outer_padding,
537                        line_number_range,
538                        source,
539                        &parts,
540                        message,
541                    )?;
542                }
543                // Additional notes, potentially with labels
544                //
545                // ```text
546                // = note: blabla is defined here
547                // | 2       let a: [bool; 2] = [true, false];
548                // |             ^
549                // ```
550                Subdiagnostic::SpannedNote(SpannedNote {
551                    severity,
552                    message,
553                    labels,
554                }) => {
555                    let (outer_padding, labeled_files) = self.label_files(files, labels)?;
556                    renderer.render_header(None, *severity, None, message.as_str())?;
557                    self.render_source_snippets(outer_padding, files, labeled_files, renderer)?;
558                }
559            }
560        }
561
562        if multiline_suggestions > 0 {
563            renderer.render_snippet_note(
564                outer_padding,
565                &format!(
566                    "(Note: ignored {} multi-line suggestion{})",
567                    multiline_suggestions,
568                    if multiline_suggestions == 1 { "" } else { "s" }
569                ),
570            )?;
571        }
572
573        if empty_suggestions > 0 {
574            renderer.render_snippet_note(
575                outer_padding,
576                &format!(
577                    "(Note: ignored {} empty suggestion{})",
578                    empty_suggestions,
579                    if empty_suggestions == 1 { "" } else { "s" },
580                ),
581            )?;
582        }
583
584        renderer.render_empty()
585    }

588/// Output a short diagnostic, with a line number, severity, and message.
589pub struct ShortDiagnostic<'diagnostic, FileId> {
590    diagnostic: &'diagnostic Diagnostic<FileId>,
591    show_notes: bool,
592}
593
594impl<'diagnostic, FileId> ShortDiagnostic<'diagnostic, FileId>
595where
596    FileId: Copy + PartialEq,
597{
598    pub fn new(
599        diagnostic: &'diagnostic Diagnostic<FileId>,
600        show_notes: bool,
601    ) -> ShortDiagnostic<'diagnostic, FileId> {
602        ShortDiagnostic {
603            diagnostic,
604            show_notes,
605        }
606    }
607
608    pub fn render<'files>(
609        &self,
610        files: &'files impl Files<'files, FileId = FileId>,
611        renderer: &mut Renderer<'_, '_>,
612    ) -> Result<(), Error>
613    where
614        FileId: 'files,
615    {
616        // Located headers
617        //
618        // ```text
619        // test:2:9: error[E0001]: unexpected type in `+` application
620        // ```
621        let mut primary_labels_encountered = 0;
622        let labels = self.diagnostic.labels.iter();
623        for label in labels.filter(|label| label.style == LabelStyle::Primary) {
624            primary_labels_encountered += 1;
625
626            renderer.render_header(
627                Some(&Locus {
628                    name: files.name(label.file_id)?.to_string(),
629                    location: files.location(label.file_id, label.range.start)?,
630                }),
631                self.diagnostic.severity,
632                self.diagnostic.code.as_deref(),
633                self.diagnostic.message.as_str(),
634            )?;
635        }
636
637        // Fallback to printing a non-located header if no primary labels were encountered
638        //
639        // ```text
640        // error[E0002]: Bad config found
641        // ```
642        if primary_labels_encountered == 0 {
643            renderer.render_header(
644                None,
645                self.diagnostic.severity,
646                self.diagnostic.code.as_deref(),
647                self.diagnostic.message.as_str(),
648            )?;
649        }
650
651        if self.show_notes {
652            // Additional notes
653            //
654            // ```text
655            // = expected type `Int`
656            //      found type `String`
657            // ```
658            for note in &self.diagnostic.notes {
659                renderer.render_snippet_note(0, note)?;
660            }
661        }
662
663        Ok(())
664    }
