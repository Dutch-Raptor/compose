13/// A managed heap containing [`HeapItem`]s accessible through [`UntypedRef`] or [`HeapRef`] keys.
14///
15/// The heap implements garbage collection through a simple mark and sweep algorithm. When cleaning the roots need to be provided.
16/// These roots are any type implementing the [Trace] trait. Anything not accessible through the provided roots will be deallocated.
17///
18/// # Example: Interacting with heap values
19///
20/// ```
21/// use compose_library::{gc::Heap, Value, gc::HeapItem};
22/// let mut heap = Heap::new();
23///
24/// let value = Value::Int(6);
25///
26/// // Allocate a value on the heap and get a key to it.
27/// let key = heap.alloc(value);
28///
29/// // Now the value is accessible through the key.
30/// assert_eq!(heap.get(key), Some(&Value::Int(6)));
31///
32/// let value_mut = heap.get_mut(key).unwrap();
33/// *value_mut = Value::Int(7);
34///
35/// // The value is updated and anyone with the key will see the new value
36/// assert_eq!(heap.get(key), Some(&Value::Int(7)));
37///
38/// // Remove the value from the heap. This will give ownership of the value to the caller.
39/// let removed_item = heap.remove(key);
40///
41/// assert_eq!(removed_item, Some(HeapItem::Value(Value::Int(7))));
42/// assert_eq!(heap.get(key), None);
43/// ```
44///
45/// # Example: Garbage Collection
46///
47/// ```
48/// use compose_library::gc::{Heap, UntypedRef};
49/// use compose_library::Value;
50/// let mut heap = Heap::new();
51///
52/// let key = heap.alloc(Value::Int(6));
53/// assert_eq!(heap.metadata().heap_size, 1);
54/// assert_eq!(heap.get(key), Some(&Value::Int(6)));
55///
56/// // clean while passing the key to the value as a simple root
57/// heap.clean(&key);
58///
59/// // The value is still accessible and the heap size hasn't changed
60/// assert_eq!(heap.metadata().heap_size, 1);
61/// assert_eq!(heap.get(key), Some(&Value::Int(6)));
62///
63/// // clean while passing a root that doesn't have access to the value
64/// heap.clean(&None::<UntypedRef>);
65///
66/// assert_eq!(heap.metadata().heap_size, 0);
67/// assert_eq!(heap.get(key), None);
68/// ```
69#[derive(Debug)]
70pub struct Heap {
71    map: SlotMap<UntypedRef, HeapItem>,
72    policy: Box<dyn GcTriggerPolicy>,
73}

154impl Heap {
155    pub fn new() -> Self {
156        Self {
157            map: SlotMap::with_key(),
158            policy: Box::new(SimplePolicy {
159                heap_size_threshold: 500,
160            }),
161        }
162    }
163
164    pub fn get_mut<T: HeapObject>(&mut self, key: HeapRef<T>) -> Option<&mut T> {
165        self.map.get_mut(*key)?.as_type_mut()
166    }
167    pub fn alloc<T: HeapObject>(&mut self, value: T) -> HeapRef<T> {
168        self.map.insert(value.to_untyped()).into()
169    }
170
171    pub fn metadata(&self) -> GcData {
172        GcData {
173            heap_size: self.map.len(),
174        }
175    }
176
177    pub fn get<T: HeapObject>(&self, key: HeapRef<T>) -> Option<&T> {
178        self.map.get(*key)?.as_type()
179    }
180
181    pub fn remove<T: HeapObject>(&mut self, key: HeapRef<T>) -> Option<HeapItem> {
182        self.map.remove(*key)
183    }
184
185    pub fn get_untyped(&self, key: UntypedRef) -> Option<&HeapItem> {
186        self.map.get(key)
187    }
188}

190/// A type that can keep references to heap values
191pub trait Trace: Debug {
192    fn visit_refs(&self, f: &mut dyn FnMut(UntypedRef));
193}
194
195impl Trace for &[UntypedRef] {
196    fn visit_refs(&self, f: &mut dyn FnMut(UntypedRef)) {
197        for key in *self {
198            f(*key)
199        }
200    }
201}
202
203impl<T: HeapObject> Trace for HeapRef<T> {
204    fn visit_refs(&self, f: &mut dyn FnMut(UntypedRef)) {
205        f(self.key())
206    }
207}
208
209impl Trace for UntypedRef {
210    fn visit_refs(&self, f: &mut dyn FnMut(UntypedRef)) {
211        f(*self)
212    }
213}
214
215impl<T: Trace> Trace for Option<T> {
216    fn visit_refs(&self, f: &mut dyn FnMut(UntypedRef)) {
217        if let Some(v) = self {
218            v.visit_refs(f)
219        }
220    }
221}

223pub trait HeapObject: Trace + Debug {
224    fn from_untyped(untyped: &HeapItem) -> Option<&Self>;
225    fn from_untyped_mut(untyped: &mut HeapItem) -> Option<&mut Self>;
226    fn to_untyped(self) -> HeapItem;
227}

229macro_rules! heap_enum {
230    (
231        $(#[$meta:meta])*
232        pub enum $enum_name:ident {
233            $(
234                $variant:ident($ty:ty)
235            ),* $(,)?
236        }
237    ) => {
238        $(#[$meta])*
239        pub enum $enum_name {
240            $(
241                $variant($ty),
242            )*
243        }
244
245        impl Trace for $enum_name {
246            fn visit_refs(&self, f: &mut dyn FnMut(UntypedRef)) {
247                match self {
248                    $(
249                        $enum_name::$variant(inner) => inner.visit_refs(f),
250                    )*
251                }
252            }
253        }
254    };
255}
256
257impl Trace for Value {
258    fn visit_refs(&self, f: &mut dyn FnMut(UntypedRef)) {
259        match self {
260            Value::Int(_) => {}
261            Value::Bool(_) => {}
262            Value::Unit(_) => {}
263            Value::Str(_) => {}
264            Value::Func(func) => func.visit_refs(f),
265            Value::Type(ty) => ty.visit_refs(f),
266            Value::Iterator(i) => i.visit_refs(f),
267            Value::Box(b) => f(b.key()),
268            Value::Array(a) => a.visit_refs(f),
269            Value::Range(r) => r.visit_refs(f),
270            Value::Map(m) => m.visit_refs(f),
271            Value::Module(m) => m.visit_refs(f),
272        }
273    }
274}

276macro_rules! impl_heap_obj {
277    ($ident:ident, $ty:ty) => {
278        impl HeapObject for $ty {
279            fn from_untyped(untyped: &HeapItem) -> Option<&Self> {
280                match untyped {
281                    HeapItem::$ident(v) => Some(v),
282                    _ => None,
283                }
284            }
285
286            fn from_untyped_mut(untyped: &mut HeapItem) -> Option<&mut Self> {
287                match untyped {
288                    HeapItem::$ident(v) => Some(v),
289                    _ => None,
290                }
291            }
292
293            fn to_untyped(self) -> HeapItem {
294                HeapItem::$ident(self)
295            }
296        }
297    };
298}
