1use compose_error_codes::E0012_PREDICATE_MUST_RETURN_BOOLEAN;
2use compose_library::diag::{SourceResult, bail, error};
3use std::iter;
4use compose_library::foundations::types::Func;
5use compose_library::{Value, Vm};
6use compose_library::foundations::args::Args;

8pub fn eval_predicate(
9    vm: &mut dyn Vm,
10    predicate: &Func,
11    value: Value,
12    callee: &str,
13) -> SourceResult<bool> {
14    let span = predicate.span;
15    let args = Args::new(span, iter::once(value));
16    match predicate.call(vm, args)? {
17        Value::Bool(b) => Ok(b),
18        other => {
19            let mut err = error!(
20                span, "predicate must return a boolean";
21                label_message: "this function should return a boolean, but returned type `{}` instead", other.ty();
22                note: "a predicate function passed to `{callee}` must return either `true` or `false`";
23                code: &E0012_PREDICATE_MUST_RETURN_BOOLEAN;
24            );
25
26            if let Some(hint) = predicate_hint(&other) {
27                err.hint(hint);
28            }
29
30            bail!(err);
31        }
32    }
33}
34
35pub fn eval_func(vm: &mut dyn Vm, func: &Func, args: impl IntoIterator<Item = Value>) -> SourceResult<Value> {
36    func.call(vm, Args::new(func.span, args))
37}
38
39fn predicate_hint(value: &Value) -> Option<&'static str> {
40    match value {
41        Value::Int(_) => Some("did you mean to write a comparison like `x == 1` or `x > 1`?"),
42        Value::Str(_) => Some("did you mean to compare it, like `x == \"hello\"`?"),
43        Value::Array(_) => {
44            Some("did you mean to check the contents, e.g., `x.contains(...)` or `x.len() > 0`?")
45        }
46        _ => None,
47    }
48}
