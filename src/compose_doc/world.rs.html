1use std::io::{Read, Write};
2use std::sync::Mutex;
3use compose_library::diag::FileResult;
4use compose_library::{Library, World};
5use compose_syntax::{FileId, Source};

7#[derive(Debug)]
8pub(crate) struct DocWorld {
9    pub source: Source,
10    library: Library,
11    pub stdout: Mutex<String>
12}
13
14impl Clone for DocWorld {
15    fn clone(&self) -> Self {
16        Self {
17            source: self.source.clone(),
18            library: self.library.clone(),
19            stdout: Mutex::new(self.stdout.lock().expect("failed to lock stdout").clone())
20        }
21    }   
22}
23
24impl DocWorld {
25    pub(crate) fn from_str(text: &str) -> Self {
26        let entrypoint = FileId::new("main.comp");
27        let source = Source::new(entrypoint, text.to_string());
28
29        Self {
30            source,
31            library: Library::default(),
32            stdout: Mutex::new(String::new())
33        }
34    }
35}
36
37impl World for DocWorld {
38    fn entry_point(&self) -> FileId {
39        self.source.id()
40    }
41
42    fn source(&self, _file_id: FileId) -> FileResult<Source> {
43        // Just return the source, no other files are needed.
44        Ok(self.source.clone())
45    }
46
47    fn library(&self) -> &Library {
48        &self.library
49    }
50
51    fn write(&self, f: &mut dyn FnMut(&mut dyn Write) -> std::io::Result<()>) -> std::io::Result<()> {
52        let mut buffer: Vec<u8> = Vec::new();
53        f(&mut buffer)?;
54        let output = String::from_utf8(buffer).expect("Invalid UTF-8");
55        self.stdout.lock().expect("failed to lock stdout").push_str(&output);
56        Ok(())
57    }
58
59    fn read(&self, f: &mut dyn FnMut(&mut dyn Read) -> std::io::Result<()>) -> std::io::Result<()> {
60        f(&mut std::io::stdin())
61    }
